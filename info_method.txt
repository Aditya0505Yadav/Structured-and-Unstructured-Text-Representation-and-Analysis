    def info(self, table: Optional[str] = None):
        """
        Show complete information about database and tables.
        Displays: tables list, schema, sample data, and statistics.
        
        Args:
            table: Specific table (shows all if None)
        
        Example:
            >>> sutra.info()  # Show everything
            >>> sutra.info("sales_data")  # Show specific table
        """
        if not self.schema:
            self._refresh_schema()
        
        tables_to_show = [table] if table else list(self.schema.keys())
        
        print("\n" + "="*80)
        print("ğŸ“Š DATABASE INFORMATION")
        print("="*80)
        
        # Show all tables
        all_tables = self.tables()
        print(f"\nğŸ“ Tables ({len(all_tables)}):")
        for i, tbl in enumerate(all_tables, 1):
            # Get row count
            count_query = f"SELECT COUNT(*) FROM {tbl}"
            row_count = pd.read_sql_query(count_query, self.conn).iloc[0, 0]
            col_count = len(self.schema.get(tbl, {}))
            
            marker = "ğŸ‘‰" if tbl == self.current_table else "  "
            print(f"{marker} {i}. {tbl:<30} ({row_count} rows, {col_count} columns)")
        
        # Show detailed info for each table
        for tbl in tables_to_show:
            if tbl not in self.schema:
                continue
                
            print("\n" + "-"*80)
            print(f"ğŸ“‹ Table: {tbl}")
            print("-"*80)
            
            # Schema
            print("\nğŸ”¤ Columns:")
            for col, dtype in self.schema[tbl].items():
                print(f"   â€¢ {col:<30} ({dtype})")
            
            # Sample data
            sample_df = pd.read_sql_query(f"SELECT * FROM {tbl} LIMIT 3", self.conn)
            print(f"\nğŸ“Š Sample Data (first 3 rows):")
            print(sample_df.to_string(index=False))
            
            # Statistics for numeric columns
            try:
                stats_df = pd.read_sql_query(f"SELECT * FROM {tbl}", self.conn)
                numeric_cols = stats_df.select_dtypes(include=[np.number]).columns.tolist()
                
                if numeric_cols:
                    print(f"\nğŸ“ˆ Statistics:")
                    for col in numeric_cols[:5]:  # Show first 5 numeric columns
                        min_val = stats_df[col].min()
                        max_val = stats_df[col].max()
                        avg_val = stats_df[col].mean()
                        print(f"   {col}: min={min_val}, max={max_val}, avg={avg_val:.2f}")
            except:
                pass
        
        print("\n" + "="*80)
        return self
